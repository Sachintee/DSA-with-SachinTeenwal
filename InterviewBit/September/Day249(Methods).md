## Code(C++)
```cpp
#include <iostream>
using namespace std;

// Function to add two integers
int add(int a, int b) {
    return a + b;
}

// Function to multiply two integers
int multiply(int a, int b) {
    return a * b;
}

int main() {
    int a, b;
    cin >> a >> b;
    cout << add(a, b) << endl;
    cout << multiply(a, b) << endl;
    return 0;
}

```
